7.2 Clase ResourceManager (Core)
class_name ResourceManager
extends Node
signal resource_changed(resource_type: String, amount: float)
signal capacity_warning(resource_type: String)
enum ResourceType {ESPECIA, SOLARIS, AGUA, MATERIALES, CONOCIMIENTO}
@export var resources: Dictionary = {
ResourceType.ESPECIA: 0.0,
ResourceType.SOLARIS: 100.0,
ResourceType.AGUA: 200.0,
ResourceType.MATERIALES: 50.0,
ResourceType.CONOCIMIENTO: 0.
}
@export var capacities: Dictionary = {
ResourceType.ESPECIA: 100.0,
ResourceType.SOLARIS: 500.0,
ResourceType.AGUA: 1000.0,
ResourceType.MATERIALES: 200.0,
ResourceType.CONOCIMIENTO: 50.
}
func modify_resource(type: ResourceType, amount: float) -> bool:
var new_value = resources[type] + amount
var capacity = capacities[type]
if amount > 0 and new_value > capacity:
resource_changed.emit(type, capacity - resources[type])
resources[type] = capacity
capacity_warning.emit(type)
return false
resources[type] = clamp(new_value, 0, capacity)
resource_changed.emit(type, amount)
return true
func get_resource(type: ResourceType) -> float:
return resources[type]
func get_capacity(type: ResourceType) -> float:
return capacities[type]
func get_fill_percentage(type: ResourceType) -> float:
return resources[type] / capacities[type]

Cargar datos del recurso de instalaciГіn
var data = load_installation_data(installation_type)
base_production = data["production"]
maintenance_cost = data["maintenance"]

func process_production() -> Dictionary:
if not is_active:
return {}

var output = {}
for resource_type in base_production.keys():
var amount = base_production[resource_type] * efficiency
output[resource_type] = amount

current_round_production = output.values().reduce(func(a, b): return a + b, 0.0)
production_complete.emit(output)
return output

func apply_event(event_type: String, intensity: float) -> void:
match event_type:
"storm":
efficiency = clamp(efficiency * (1.0 - intensity), 0.1, 1.0)
"sandworm":
efficiency = clamp(efficiency * (1.0 + intensity), 0.5, 2.0)
"malfunction":
efficiency = clamp(efficiency - intensity, 0.1, 1.0)

efficiency_changed.emit(efficiency)

func get_maintenance_cost() -> Dictionary:
return maintenance_cost.duplicate()

func upgrade() -> void:
efficiency = clamp(efficiency + 0.1, 1.0, 2.0)
func destroy() -> void:
queue_free()

[resource_type])
